tryIsLeapYear Enclosing


function isLeapYear(year) {
    if (year === undefined) {
        throw new Error("Missing argument year error");
    }
    if (!Number.isInteger(year) && year !== 0) {
        throw new Error("Non-integer argument year error");
    }
    // Leap year logic
    if (year % 4 === 0) {
        if (year % 100 !== 0 || year % 400 === 0) {
            return true;
        }
    }
    return false;
}

function tryIsLeapYear(year) {
    try {
        return isLeapYear(year);
    } catch (e) {
        console.error("Error in isLeapYear:", e.message);
        return false; 
    }
}

console.log("tryIsLeapYear(2024):", tryIsLeapYear(2024));
console.log("tryIsLeapYear():", tryIsLeapYear());
console.log("tryIsLeapYear(2024.5):", tryIsLeapYear(2024.5));

The isLeapYear function now throws errors directly for invalid input. The tryIsLeapYear function encloses the call to isLeapYear within a try...catch block. If isLeapYear throws an error, the catch block in tryIsLeapYear catches it, logs the error message to the console, and returns false (or handles the error as needed).
