Error Handling in Enclosing Function


function isLeapYear(year) {
    // Leap year logic - no error throwing here
    if (year % 4 === 0) {
        if (year % 100 !== 0 || year % 400 === 0) {
            return true;
        }
    }
    return false;
}

function tryIsLeapYear(year) {
    try {
        if (year === undefined) {
            throw new Error("Missing argument year error");
        }
        if (!Number.isInteger(year) && year !== 0) {
            throw new Error("Non-integer argument year error");
        }
        return isLeapYear(year);
    } catch (e) {
        console.error("Error in tryIsLeapYear:", e.message);
        return false;
    }
}

console.log("tryIsLeapYear(2024):", tryIsLeapYear(2024));
console.log("tryIsLeapYear():", tryIsLeapYear());
console.log("tryIsLeapYear(2024.5):", tryIsLeapYear(2024.5));

Now, the isLeapYear function only contains the core leap year calculation logic and does not throw any errors. The tryIsLeapYear function is responsible for both checking the validity of the input year and throwing errors if the argument is missing or not an integer. It also encloses the call to isLeapYear in a try...catch block to handle any potential errors (although in this specific case, isLeapYear itself doesn't throw errors based on input).
